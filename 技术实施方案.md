# 校园二手物品交易平台技术实施方案

## 一、系统架构设计

### 1.1 技术选型

| 技术层次 | 选型方案 | 说明 |
|---------|---------|------|
| 前端框架 | 原生微信小程序 | 无额外学习成本，文档完善 |
| 后端服务 | 微信云开发 | 免费额度充足，零运维 |
| 数据库 | 云数据库 NoSQL | 支持 JSON 文档存储 |
| 函数执行 | 云函数 | Node.js 运行环境 |
| 文件存储 | 云存储 | 商品图片存储 |

### 1.2 系统架构图

```
┌─────────────────────────────────────────────────────┐
│                    微信小程序                         │
├─────────────┬─────────────┬─────────────┬───────────┤
│   用户页    │   商品页    │   交易页    │   消息页   │
└──────┬──────┴──────┬──────┴──────┬──────┴─────┬───┘
       │             │             │            │
       └─────────────┴─────────────┴────────────┘
                     │ 微信 API
       ┌─────────────┴─────────────┐
       │      云开发后端           │
       ├────────┬────────┬───────┤
       │云函数   │云数据库│云存储 │
       └────────┴────────┴───────┘
```

### 1.3 数据流设计

```
用户操作 → 小程序页面 → wx.cloud 调用 → 云函数/云数据库 → 返回结果 → 页面更新
```

### 2.1 数据集合总览

| 集合名称 | 用途 | 主键 |
|---------|------|------|
| users | 用户信息 | _id |

### 2.2 详细字段设计

#### 2.2.1 users（用户集合）

```javascript
{
  _id: ObjectId,           // 自动生成
  openid: String,          // 微信授权唯一标识 [必须]
  phone: String,           // 手机号 [普通用户必填]
  nickname: String,       // 昵称
  avatar: String,         // 头像URL
  role: String,           // 角色: "admin" | "user"
  createdAt: Date,       // 注册时间
  updatedAt: Date        // 更新时间
}
```

**索引**：
- phone: 唯一索引（普通用户）
- openid: 唯一索引

#### 2.2.2 goods（商品集合）

```javascript
{
  _id: ObjectId,           // 自动生成
  sellerId: ObjectId,     // 卖家ID [必须]
  category: String,        // 分类 [书籍|数码|服饰|生活用品] [必须]
  title: String,          // 标题 1-50字符 [必须]
  description: String,    // 描述 1-2000字符 [必须]
  price: Number,          // 价格 0.01-9999 [必须]
  images: String[],       // 图片URL数组 最多6张
  status: String,        // 状态 [上架中|已下架|已售出]
  createdAt: Date,      // 发布时间
  updatedAt: Date       // 更新时间
}
```

**索引**：
- sellerId: 普通索引
- category: 普通索引
- status: 普通索引
- createdAt: 降序索引（默认排序）

#### 2.2.3 orders（订单集合）

```javascript
{
  _id: ObjectId,           // 自动生成
  orderNo: String,         // 订单号 [必须唯一]
  goodsId: ObjectId,      // 商品ID [必须]
  buyerId: ObjectId,      // 买家ID [必须]
  sellerId: ObjectId,     // 卖家ID [必须]
  receiverName: String,  // 收货人姓名 [必须]
  receiverPhone: String, // 收货人电话 [必须]
  receiverAddress: String, // 收货地址 [必须]
  totalPrice: Number,    // 订单金额 [必须]
  status: String,        // 状态 [待发货|已发货|已完成]
  createdAt: Date,       // 创建时间
  updatedAt: Date       // 更新时间
}
```

**索引**：
- orderNo: 唯一索引
- goodsId: 普通索引
- buyerId: 普通索引
- sellerId: 普通索引

#### 2.2.4 notices（通知集合）

```javascript
{
  _id: ObjectId,           // 自动生成
  userId: ObjectId,       // 接收用户ID [必须]
  type: String,           // 通知类型 [订单状态变更|系统通知]
  title: String,         // 通知标题
  content: String,      // 通知内容
  relatedId: ObjectId,   // 关联ID（订单/商品）

}
```

**索引**：
- userId: 普通索引
- isRead: 普通索引

### 3.2 核心云函数实现

#### 3.2.1 user/login（微信登录）

```javascript
// cloudfunctions/user/login/index.js
const cloud = require('wx-server-sdk')
cloud.init()

exports.main = async (event, context) => {
  const wxContext = cloud.getWXContext()
  const openid = wxContext.OPENID
  
  const db = cloud.database()
  const _ = db.command
  
  // 查询是否已存在
  let user = await db.collection('users').where({
    openid: openid
  }).get()
  
  if (user.data.length === 0) {
    // 新用户自动创建
    await db.collection('users').add({
      data: {
        openid: openid,
        nickname: '用户' + openid.slice(-4),
        role: 'user',
        createdAt: new Date(),
        updatedAt: new Date()
      }
    })
    user = await db.collection('users').where({ openid: openid }).get()
  }
  
  return {
    success: true,
    user: user.data[0]
  }
}
```

#### 3.2.2 user/register（用户注册）

```javascript
// cloudfunctions/user/register/index.js
const cloud = require('wx-server-sdk')
cloud.init()

exports.main = async (event, context) => {
  const { phone, password } = event
  const wxContext = cloud.getWXContext()
  const openid = wxContext.OPENID
  
  if (!phone || !password) {
    return { success: false, error: '手机号和密码不能为空' }
  }
  
  const db = cloud.database()
  
  // 检查手机号是否已注册
  const exist = await db.collection('users').where({
    phone: phone
  }).get()
  
  if (exist.data.length > 0) {
    return { success: false, error: '该手机号已注册' }
  }
  
  // 更新用户信息
  const result = await db.collection('users').where({
    openid: openid
  }).update({
    data: {
      phone: phone,
      password: password, // 生产环境需加密存储
      role: 'user',
      updatedAt: new Date()
    }
  })
  
  return { success: true }
}
```

#### 3.2.3 goods/publish（发布商品）

```javascript
// cloudfunctions/goods/publish/index.js
const cloud = require('wx-server-sdk')
cloud.init()

exports.main = async (event, context) => {
  const { category, title, description, price, images } = event
  const wxContext = cloud.getWXContext()
  const openid = wxContext.OPENID
  
  // 参数校验
  if (!category || !title || !description || !price) {
    return { success: false, error: '缺少必填参数' }
  }
  
  const db = cloud.database()
  
  // 获取卖家ID
  const seller = await db.collection('users').where({
    openid: openid
  }).get()
  
  if (seller.data.length === 0) {
    return { success: false, error: '用户不存在' }
  }
  
  // 创建商品
  const result = await db.collection('goods').add({
    data: {
      sellerId: seller.data[0]._id,
      category,
      title,
      description,
      price: Number(price),
      images: images || [],
      status: '上架中',
      createdAt: new Date(),
      updatedAt: new Date()
    }
  })
  
  return { success: true, goodsId: result._id }
}
```

#### 3.2.4 goods/getList（获取商品列表）

```javascript
// cloudfunctions/goods/getList/index.js
const cloud = require('wx-server-sdk')
cloud.init()

exports.main = async (event, context) => {
  const { 
    page = 1, 
    pageSize = 10, 
    category, 
    status = '上架中',
    sortBy = 'createdAt',
    sortOrder = 'desc'
  } = event
  
  const db = cloud.database()
  const _ = db.command
  
  // 构建查询条件
  let query = { status }
  if (category) {
    query.category = category
  }
  
  // 构建排序
  let order = {}
  order[sortBy] = sortOrder === 'asc' ? 1 : -1
  
  // 分页查询
  const skip = (page - 1) * pageSize
  
  const result = await db.collection('goods')
    .where(query)
    .orderBy(sortBy, sortOrder)
    .skip(skip)
    .limit(pageSize)
    .get()
  
  // 获取总数
  const count = await db.collection('goods')
    .where(query)
    .count()
  
  return {
    success: true,
    list: result.data,
    total: count.total,
    page,
    pageSize
  }
}
```

#### 3.2.5 goods/search（搜索商品）

```javascript
// cloudfunctions/goods/search/index.js
const cloud = require('wx-server-sdk')
cloud.init()

exports.main = async (event, context) => {
  const { keyword, page = 1, pageSize = 10 } = event
  
  if (!keyword) {
    return { success: false, error: '关键词不能为空' }
  }
  
  const db = cloud.database()
  const _ = db.command
  
  // 使用正则搜索标题
  const result = await db.collection('goods')
    .where(_.or([
      { title: db.RegExp({ regex: keyword, options: 'i' }) }
    ]))
    .orderBy('createdAt', 'desc')
    .skip((page - 1) * pageSize)
    .limit(pageSize)
    .get()
  
  return {
    success: true,
    list: result.data,
    keyword
  }
}
```

#### 3.2.6 order/create（创建订单）

```javascript
// cloudfunctions/order/create/index.js
const cloud = require('wx-server-sdk')
cloud.init()

exports.main = async (event, context) => {
  const { goodsId, receiverName, receiverPhone, receiverAddress } = event
  const wxContext = cloud.getWXContext()
  const openid = wxContext.OPENID
  
  const db = cloud.database()
  
  // 获取商品信息
  const goods = await db.collection('goods').doc(goodsId).get()
  
  if (!goods.data) {
    return { success: false, error: '商品不存在' }
  }
  
  if (goods.data.status !== '上架中') {
    return { success: false, error: '商品已下架或已售出' }
  }
  
  // 获取买家信息
  const buyer = await db.collection('users').where({
    openid: openid
  }).get()
  
  // 生成订单号
  const orderNo = 'ORD' + Date.now() + Math.random().toString(36).slice(-4)
  
  // 创建订单
  const orderResult = await db.collection('orders').add({
    data: {
      orderNo,
      goodsId,
      buyerId: buyer.data[0]._id,
      sellerId: goods.data.sellerId,
      receiverName,
      receiverPhone,
      receiverAddress,
      totalPrice: goods.data.price,
      status: '待发货',
      createdAt: new Date(),
      updatedAt: new Date()
    }
  })
  
  // 更新商品状态为已售出
  await db.collection('goods').doc(goodsId).update({
    data: {
      status: '已售出',
      updatedAt: new Date()
    }
  })
  
  return { success: true, orderId: orderResult._id }
}
```

## 六、安全策略

### 6.1 用户鉴权

- 使用微信 `openid` 作为用户唯一标识
- 普通用户需绑定手机号 + 密码登录
- 管理员使用固定账号（手机号：17359050190，密码：123456）
- 云函数入口处验证 `wxContext.OPENID`

### 6.2 数据权限

| 操作 | 权限要求 |
|------|---------|
| 查看商品 | 所有人可访问 |
| 发布商品 | 登录用户 |
| 修改自己商品 | 商品所有者 |
| 删除任意商品 | 管理员 |
| 查看自己订单 | 订单相关方 |
| 管理用户 | 管理员 |

### 6.3 接口安全

- 云函数配置 "登录检查" 白名单
- 敏感操作需验证 openid 匹配
- 商品价格等数值在后端校验
